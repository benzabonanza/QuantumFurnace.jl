struct TrajectoryScratch{T<:Complex}
    jump_oft::Matrix{T}
    psi_tmp::Vector{T}
    Rpsi::Vector{T}
    rho_acc::Matrix{T}
end

function TrajectoryScratch(::Type{CT}, dim::Int) where {CT<:Complex}
    TrajectoryScratch{CT}(
        zeros(CT, dim, dim),  # jump_oft
        zeros(CT, dim),       # psi_tmp
        zeros(CT, dim),       # Rpsi
        zeros(CT, dim, dim),  # rho_acc
    )
end

Note: this is a NEW method of the `Workspace` outer constructor, dispatching on the Config type. It is NOT a second constructor for the same variant -- it is the ONE constructor for the Trajectory variant.

   The constructor should:
   - Accept the same inputs as `build_trajectoryframework` + `_build_framework_and_seed`, but consolidated into one call
   - Actually, to maintain backward compatibility with callers that pass precomputed_data and scratch, provide a workspace constructor that takes `(config, hamiltonian, jumps; trotter, delta)` and does the full setup internally:

     - Call `_precompute_data(config, ham_or_trott)`
     - Build per-operator data (R^a, K0^a, U_residual^a, U_B^a) -- same logic as current `build_trajectoryframework`
     - Store as flat vectors: `Rs::Vector{Matrix{CT}}`, `K0s::Vector{Matrix{CT}}`, `U_residuals::Vector{Matrix{CT}}`, `U_Bs::Vector{Union{Nothing, Matrix{CT}}}`
     - Compute hot-path scalars (scaled_prefactor, alpha, etc.)
     - Build TrajectoryScratch
     - Return `Workspace{Trajectory,D,C,T}(...)`

   **Fields on Workspace{Trajectory,...}** (stored as Workspace fields, using Union{Nothing,...} for fields not used by other variants):
   - `jumps`: `Vector{JumpOp{Matrix{CT}}}` (concrete-typed)
   - `ham_or_trott`: stored for EnergyDomain `step_along_trajectory!` (needs `ham.bohr_freqs`)
   - `n_jumps`: Int
   - `delta`: Float64
   - `alpha`: Float64
   - `scaled_prefactor`: Float64
   - `sigma`: Float64
   - `transition`: concrete closure
   - `energy_labels`: Vector{Float64}
   - `oft_nufft_prefactors`: NUFFTPrefactors or Nothing
   - `Rs`: Vector{Matrix{CT}} -- per-jump R^a
   - `K0s`: Vector{Matrix{CT}} -- per-jump K0^a
   - `U_residuals`: Vector{Matrix{CT}} -- per-jump U_residual^a
   - `U_Bs`: Vector{Union{Nothing, Matrix{CT}}} -- per-jump coherent unitary
   - `scratch`: TrajectoryScratch{CT}

   **Dead fields NOT included** (removed vs TrajectoryFramework):
   - `config` -- callers have it; not accessed in hot path
   - `precomputed_data` -- absorbed into flat fields
   - `delta_eff` -- always equals `delta`; use `delta` directly
   - `domain` -- available via type parameter D

   **NOTE on `ham_or_trott`**: This needs to be stored for EnergyDomain step_along_trajectory! which accesses `ham.bohr_freqs`. For Time/TrotterDomain it is not needed in the hot path. Store it as a workspace field.

   - Return `(ws, actual_seed)` where `ws` is `Workspace{Trajectory,...}`

**IMPORTANT**: The current code uses `KrausScratch` as builder scratch within `build_trajectoryframework` to compute per-operator R via `_precompute_R`. The new Workspace{Trajectory} constructor must either:
