Create `test/test_allocation.jl` with @allocated tests for each optimized function. Follow the warmup pattern from 11-RESEARCH.md: call the function once to trigger JIT, then measure with @allocated.

**Test structure:**

```julia
@testset "Allocation Regression" begin

    @testset "B_bohr allocations" begin
        # Setup: need a config, hamiltonian, jump for BohrDomain
        config = make_liouv_config(BohrDomain())
        jump = TEST_JUMPS[1]

        # Test single-jump variant
        B_ref = QuantumFurnace.B_bohr(TEST_HAM, jump, config)  # warmup
        allocs = @allocated QuantumFurnace.B_bohr(TEST_HAM, jump, config)
        # B_bohr allocates its return value B (dim x dim) and f_A_nu_1 buffer (dim x dim).
        # Total expected: 2 * DIM^2 * sizeof(ComplexF64) + small overhead.
        # It should NOT scale with number of unique Bohr frequencies.
        max_expected = 2 * DIM^2 * sizeof(ComplexF64) + 4096  # 2 matrices + overhead
        @test allocs <= max_expected

        # Test multi-jump variant
        B_ref_multi = QuantumFurnace.B_bohr(TEST_HAM, TEST_JUMPS, config)  # warmup
        allocs_multi = @allocated QuantumFurnace.B_bohr(TEST_HAM, TEST_JUMPS, config)
        # Multi-jump: B (dim x dim) + one f_A_nu_1 per jump (but it's re-allocated inside the loop
        # per jump -- that's acceptable, it's per-jump not per-frequency).
        # Actually, the multi-jump variant allocates f_A_nu_1 inside the jump loop.
        # Expected: B + num_jumps * f_A_nu_1 = (1 + num_jumps) * DIM^2 * sizeof(ComplexF64) + overhead
        num_jumps = length(TEST_JUMPS)
        max_expected_multi = (1 + num_jumps) * DIM^2 * sizeof(ComplexF64) + 4096
        @test allocs_multi <= max_expected_multi
    end

    @testset "B_time allocations" begin
        config = make_liouv_config(TimeDomain())
        precomputed = QuantumFurnace._precompute_data(TEST_HAM, config)
        (; b_minus, b_plus) = precomputed
        jump = TEST_JUMPS[1]

        # Single-jump warmup + measure
        B_ref = QuantumFurnace.B_time(jump, TEST_HAM, b_minus, b_plus, config.t0, config.beta, config.sigma)
        allocs = @allocated QuantumFurnace.B_time(jump, TEST_HAM, b_minus, b_plus, config.t0, config.beta, config.sigma)
        # Expected allocations: pre-allocated buffers (diag_u, diag_u2, b_plus_summand, tmp, B)
        # = 2 vectors (d) + 3 matrices (d x d) + scratch matrix for mul! + return value
        d = DIM
        max_expected = 10 * d^2 * sizeof(ComplexF64) + 4096
        @test allocs <= max_expected

        # Multi-jump warmup + measure
        B_ref_m = QuantumFurnace.B_time(TEST_JUMPS, TEST_HAM, b_minus, b_plus, config.t0, config.beta, config.sigma)
        allocs_m = @allocated QuantumFurnace.B_time(TEST_JUMPS, TEST_HAM, b_minus, b_plus, config.t0, config.beta, config.sigma)
        @test allocs_m <= max_expected  # same buffer structure, shared across jumps
    end

    @testset "B_trotter allocations" begin
        config = make_liouv_config(TrotterDomain())
        precomputed = QuantumFurnace._precompute_data(TEST_HAM, config; trotter=TEST_TROTTER)
        (; b_minus, b_plus) = precomputed

        # Need Trotter-basis jumps (callers now transform before calling B_trotter)
        trotter_jumps = QuantumFurnace.transform_jumps_to_basis(TEST_JUMPS, TEST_TROTTER.eigvecs)
        jump = trotter_jumps[1]

        # Single-jump warmup + measure
        B_ref = QuantumFurnace.B_trotter(jump, TEST_TROTTER, b_minus, b_plus, config.beta, config.sigma)
        allocs = @allocated QuantumFurnace.B_trotter(jump, TEST_TROTTER, b_minus, b_plus, config.beta, config.sigma)
        d = DIM
        max_expected = 10 * d^2 * sizeof(ComplexF64) + 4096
        @test allocs <= max_expected

        # Multi-jump warmup + measure
        B_ref_m = QuantumFurnace.B_trotter(trotter_jumps, TEST_TROTTER, b_minus, b_plus, config.beta, config.sigma)
        allocs_m = @allocated QuantumFurnace.B_trotter(trotter_jumps, TEST_TROTTER, b_minus, b_plus, config.beta, config.sigma)
        @test allocs_m <= max_expected
    end

    @testset "_jump_contribution! Time/Trotter filter allocation" begin
        # Verify that _jump_contribution! for Time/Trotter thermalize does not allocate
        # filter+abs vectors. This function operates in-place on scratch buffers,
        # so the only allocations should be from NUFFT prefactor lookups (Dict access).
        config_therm = make_thermalize_config(TimeDomain(), delta=0.01)
        precomputed = QuantumFurnace._precompute_data(TEST_HAM, config_therm)
        CT = ComplexF64
        d = DIM
        scratch = QuantumFurnace.KrausScratch{CT}(d)
        jump = TEST_JUMPS[1]
        evolving_dm = copy(Matrix{CT}(TEST_GIBBS))

        # Warmup
        QuantumFurnace._jump_contribution!(evolving_dm, jump, TEST_HAM, config_therm, precomputed, scratch)

        # Measure
        evolving_dm .= Matrix{CT}(TEST_GIBBS)
        allocs = @allocated QuantumFurnace._jump_contribution!(evolving_dm, jump, TEST_HAM, config_therm, precomputed, scratch)

        # _jump_contribution! should be mostly in-place. It still has mul! calls and
        # the eigen() in _finalize_kraus_step! which allocates. The key test:
        # allocations should NOT include the filter+abs vectors (2 * num_energies * sizeof(Float64)).
        # Since _finalize_kraus_step! inherently allocates (eigen decomposition), we test
        # that allocations are bounded and don't include the filter overhead.
        num_energies = length(precomputed.energy_labels)
        filter_alloc = 2 * num_energies * sizeof(Float64)  # what filter+abs WOULD allocate
        # Allow generous budget for eigen + mul! temporaries, but NOT for filter
        # The test verifies the optimization by checking allocs are below a threshold
        # that would only be exceeded if filter+abs were still present.
        # Note: we don't test for zero allocs because _finalize_kraus_step! uses eigen().
        @test allocs < 50 * d^2 * sizeof(CT)  # generous bound excluding filter overhead
    end

end
```

**Key design decisions (Claude's discretion per CONTEXT.md):**
- Use threshold-based tests (`allocs <= max_expected`) rather than exact zero, because:
  - Functions allocate their return values (B matrices)
  - Functions allocate their internal scratch buffers (diag_u, etc.)
  - `_jump_contribution!` has inherent allocations from `eigen()` in `_finalize_kraus_step!`
- The thresholds are set to catch the ELIMINATED allocations:
  - B_bohr: per-frequency sparse matrix allocations (O(num_freqs * dim^2)) would far exceed the 2-matrix threshold
  - B_time/B_trotter: per-iteration Diagonal allocations (O(num_iters * dim)) would add up
  - _jump_contribution!: filter+abs vectors are small but detectable
- Warmup call before `@allocated` to avoid JIT compilation artifacts
